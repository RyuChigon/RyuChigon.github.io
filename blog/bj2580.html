백준 2580 파이썬


문제 링크

https://www.acmicpc.net/problem/2580

풀이 방식

  1. 빈칸을 수집한다. (blanks)
  2. 빈칸을 순차적으로 valid한 값을 넣어 dfs를 한다.

코드

board = [list(map(int, input().split())) for _ in range(9)]
blanks = []
for i in range(9):
    for j in range(9):
        if board[i][j] == 0:
            blanks.append((i, j))

def is_valid(x, y, value):
    # row validation
    if value in board[x]:
        return False

    # rectangle validation
    x_range, y_range = x // 3 * 3, y // 3 * 3
    for i in range(x_range, x_range + 3):
        for j in range(y_range, y_range + 3):
            if board[i][j] and board[i][j] == value:
                return False

    # column validation
    for i in range(9):
        if board[i][y] == value:
            return False
    return True

def dfs(idx):
    if idx == len(blanks):
        for b in board:
            print(' '.join(map(str, b)))
        exit(0)
    else:
        for val in range(1, 10):
            x, y = blanks[idx]
            if is_valid(x, y, val):
                board[x][y] = val
                dfs(idx + 1)
                board[x][y] = 0
dfs(0)
